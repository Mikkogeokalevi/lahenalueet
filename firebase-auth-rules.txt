rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Vain lukuoikeus kaikille puzzles-kokoelmaan
    match /puzzles/{document} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.token.email == "toni@kauppinen.info";
    }
    
    // Estä kaikki muu pääsy
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
